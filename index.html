Batch Operations & Performance

JavaScript 
import { AdvancedRupiahFormatter } from '@devnovaa-id/new-rupiah-formatter';

const formatter = new AdvancedRupiahFormatter();

// Batch format large datasets
const productPrices = Array.from({ length: 10000 }, (_, i) => i * 1000);

// Simple batch formatting
const formattedPrices = formatter.batchFormat(productPrices);
console.log(formattedPrices.length); // 10000
console.log(formattedPrices[0]);     // "Rp1.000,00"

// Batch with progress tracking
const results = formatter.batchFormat(productPrices, {
  chunkSize: 1000,
  parallel: false,
  progress: (percent) => {
    console.log(`Progress: ${percent.toFixed(1)}%`);
  }
});

// Optimized batch processing with cache
const cachedFormatter = new AdvancedRupiahFormatter({}, ['cache']);
cachedFormatter.setCacheOptions({ maxSize: 10000 });

function processLargeDataset(dataset) {
  const start = performance.now();
  const formatted = cachedFormatter.batchFormat(dataset);
  const end = performance.now();
  
  console.log(`Processed ${dataset.length} items in ${end - start}ms`);
  return formatted;
}

// Process in chunks for memory efficiency
function chunkedBatchFormat(values, chunkSize = 1000) {
  const results = [];
  
  for (let i = 0; i < values.length; i += chunkSize) {
    const chunk = values.slice(i, i + chunkSize);
    const chunkResults = formatter.batchFormat(chunk);
    results.push(...chunkResults);
    
    // Optional: Clear memory between chunks
    if (typeof gc !== 'undefined') gc();
  }
  
  return results;
}

// Validate and format with error recovery
function safeFormat(value, options = {}) {
  const formatter = new AdvancedRupiahFormatter(options);
  const validation = formatter.validate(value);
  
  if (validation.isValid) {
    return {
      success: true,
      formatted: formatter.format(value),
      parsed: validation.value
    };
  } else {
    return {
      success: false,
      error: validation.message,
      fallback: options.fallback || 'Rp0',
      formatted: options.fallback || 'Rp0'
    };
  }
}

Memory Optimization

Optimize memory usage with proper formatter management.

Memory 
import { AdvancedRupiahFormatter, PRESETS } from '@devnovaa-id/new-rupiah-formatter';

// Singleton pattern for app-wide consistency
let sharedFormatterInstance = null;

export function getSharedFormatter() {
  if (!sharedFormatterInstance) {
    sharedFormatterInstance = new AdvancedRupiahFormatter(PRESETS.BI_STANDARD, []);
  }
  return sharedFormatterInstance;
}

// Factory pattern with caching
const formatterCache = new Map();

export function getFormatter(options = {}, plugins = []) {
  const cacheKey = JSON.stringify({ options, plugins });
  
  if (!formatterCache.has(cacheKey)) {
    const formatter = new AdvancedRupiahFormatter(options, plugins);
    formatterCache.set(cacheKey, formatter);
    
    // Limit cache size
    if (formatterCache.size > 10) {
      const firstKey = formatterCache.keys().next().value;
      formatterCache.delete(firstKey);
    }
  }
  
  return formatterCache.get(cacheKey);
}

// Lazy initialization for large applications
class FormatterService {
  constructor() {
    this._formatters = new Map();
  }
  
  getFormatter(type = 'standard') {
    if (!this._formatters.has(type)) {
      let formatter;
      switch (type) {
        case 'ecommerce':
          formatter = new AdvancedRupiahFormatter(PRESETS.ECOMMERCE, []);
          break;
        case 'accounting':
          formatter = new AdvancedRupiahFormatter(PRESETS.ACCOUNTING, ['cache']);
          break;
        default:
          formatter = new AdvancedRupiahFormatter(PRESETS.BI_STANDARD, []);
      }
      this._formatters.set(type, formatter);
    }
    return this._formatters.get(type);
  }
  
  cleanup() {
    for (const formatter of this._formatters.values()) {
      formatter.destroy();
    }
    this._formatters.clear();
  }
}

// Usage
const service = new FormatterService();
const ecommerceFormatter = service.getFormatter('ecommerce');

Batch Operations Performance

Efficiently process large datasets with optimized batch operations.

Batch Processing 
import { AdvancedRupiahFormatter } from '@devnovaa-id/new-rupiah-formatter';

// Optimized batch processing
async function optimizedBatchFormat(values, options = {}) {
  const {
    chunkSize = 1000,
    useCache = true,
    concurrency = 4
  } = options;
  
  const formatter = new AdvancedRupiahFormatter({}, useCache ? ['cache'] : []);
  
  if (values.length <= chunkSize) {
    // Single chunk processing
    return formatter.batchFormat(values);
  }
  
  // Multi-chunk processing
  const chunks = [];
  for (let i = 0; i < values.length; i += chunkSize) {
    chunks.push(values.slice(i, i + chunkSize));
  }
  
  const results = [];
  
  // Process chunks with controlled concurrency
  for (let i = 0; i < chunks.length; i += concurrency) {
    const batch = chunks.slice(i, i + concurrency);
    
    const batchResults = await Promise.all(
      batch.map(chunk => {
        // Use setTimeout to prevent blocking
        return new Promise(resolve => {
          setTimeout(() => {
            resolve(formatter.batchFormat(chunk));
          }, 0);
        });
      })
    );
    
    results.push(...batchResults.flat());
    
    // Clear memory between batches if needed
    if (typeof gc !== 'undefined') {
      gc();
    }
  }
  
  return results;
}

// Performance monitoring
function benchmarkFormatter(formatter, iterations = 10000) {
  const values = Array.from({ length: iterations }, (_, i) => i * 100);
  
  const start = performance.now();
  const results = formatter.batchFormat(values);
  const end = performance.now();
  
  const totalTime = end - start;
  const avgTime = totalTime / iterations;
  const opsPerSecond = 1000 / avgTime;
  
  return {
    totalTime,
    averageTime: avgTime,
    operationsPerSecond: opsPerSecond,
    memoryUsage: process.memoryUsage ? process.memoryUsage().heapUsed : 0
  };
}

// Compare different configurations
const configs = [
  { name: 'No plugins', plugins: [] },
  { name: 'With cache', plugins: ['cache'] },
  { name: 'With cache+compact', plugins: ['cache', 'compact'] }
];

for (const config of configs) {
  const formatter = new AdvancedRupiahFormatter({}, config.plugins);
  const stats = benchmarkFormatter(formatter, 1000);
  console.log(`${config.name}: ${stats.operationsPerSecond.toFixed(0)} ops/sec`);
}

export class MyCustomPlugin implements Plugin {
  name = 'my-plugin';
  version = '1.0.0';
  
  private config: any;
  
  constructor(config = {}) {
    this.config = config;
  }
  
  install(formatter: any, options?: any): void {
    // Add custom methods to formatter
    formatter.myCustomFormat = (value: InputValue) => {
      return `CUSTOM: ${formatter.format(value)}`;
    };
    
    // Hook into formatting process
    const originalFormat = formatter.format.bind(formatter);
    formatter.format = (value: InputValue) => {
      const result = originalFormat(value);
      return this.transformResult(result);
    };
  }
  
  uninstall(): void {
    // Clean up if needed
  }
  
  beforeFormat(value: InputValue, options: FormatOptions): InputValue | void {
    // Transform value before formatting
    if (typeof value === 'number') {
      return value * (this.config.multiplier || 1);
    }
    return value;
  }

    // Transform result after formatting
    return {
      ...result,
      formatted: `âœ¨ ${result.formatted} âœ¨`
    };
  }
  
  private transformResult(result: string): string {
    // Custom transformation logic
    return result.toUpperCase();
  }
}
