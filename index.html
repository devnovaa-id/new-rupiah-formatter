Basic Formatting

JavaScript / TypeScript 
import { formatRupiah, parseRupiah } from '@devnovaa-id/new-rupiah-formatter';

// Format numbers according to BI standard
console.log(formatRupiah(1000));           // "Rp1.000,00"
console.log(formatRupiah(1234567.89));    // "Rp1.234.567,89"
console.log(formatRupiah(-5000));         // "-Rp5.000,00"

// Parse formatted strings back to numbers
console.log(parseRupiah('Rp1.234.567,89')); // 1234567.89
console.log(parseRupiah('-Rp5.000,00'));    // -5000

// Validate Rupiah strings
import { isValidRupiah } from '@devnovaa-id/new-rupiah-formatter';
console.log(isValidRupiah('Rp1.000,00')); // true
console.log(isValidRupiah('invalid'));    // false

Input/Output Validation

JavaScript 
import { RupiahFormatter, isValidRupiah } from '@devnovaa-id/new-rupiah-formatter';

const formatter = new RupiahFormatter();

// Validate user input
function validateAndFormat(input) {
  if (!input && input !== 0) {
    return formatter.format(0);
  }
  
  // Try to parse if it's a string
  if (typeof input === 'string') {
    if (isValidRupiah(input)) {
      const parsed = formatter.parse(input);
      return formatter.format(parsed);
    } else {
      // Try to extract number from string
      const num = parseFloat(input.replace(/[^\d.,-]/g, ''));
      if (!isNaN(num)) {
        return formatter.format(num);
      }
    }
  }
  
  // If it's already a number
  if (typeof input === 'number') {
    return formatter.format(input);
  }
  
  return formatter.format(0);
}

// Examples
console.log(validateAndFormat('Rp1.000,00'));  // "Rp1.000,00"
console.log(validateAndFormat('1000'));        // "Rp1.000,00"
console.log(validateAndFormat('invalid'));     // "Rp0"
console.log(validateAndFormat(2500));          // "Rp2.500,00"
console.log(validateAndFormat(null));          // "Rp0"

Optimize Your Usage

Best practices for maximum performance and efficiency.

Instance Reuse

Reuse formatter instances instead of creating new ones for better performance.

Performance 
// ✅ GOOD: Reuse formatter instance
const formatter = new RupiahFormatter();
for (const price of prices) {
  console.log(formatter.format(price));
}

// ❌ AVOID: Creating new instances in loops
for (const price of prices) {
  const formatter = new RupiahFormatter(); // Inefficient
  console.log(formatter.format(price));
}

// ✅ Use static method for one-off formatting
console.log(RupiahFormatter.format(1000)); // "Rp1.000,00"

// ✅ Singleton pattern for app-wide consistency
const getFormatter = (() => {
  let instance;
  return () => {
    if (!instance) {
      instance = new RupiahFormatter();
    }
    return instance;
  };
})();

Batch Operations

Efficiently process large datasets with batch operations.

Batch Processing 
import { RupiahFormatter } from '@devnovaa-id/new-rupiah-formatter';

// Create single formatter instance
const formatter = new RupiahFormatter();

// Optimized batch format function
function batchFormatRupiah(values, chunkSize = 1000) {
  const results = [];
  
  for (let i = 0; i < values.length; i += chunkSize) {
    const chunk = values.slice(i, i + chunkSize);
    
    // Process chunk
    for (const value of chunk) {
      results.push(formatter.format(value));
    }
  }
  
  return results;
}

// Process in chunks for very large datasets
function chunkFormatRupiah(values, chunkSize = 1000) {
  const chunks = [];
  for (let i = 0; i < values.length; i += chunkSize) {
    chunks.push(values.slice(i, i + chunkSize));
  }
  
  return chunks.flatMap(chunk => batchFormatRupiah(chunk));
}

// Usage with large dataset
const largeDataset = Array.from({ length: 10000 }, (_, i) => i * 1000);
const formatted = batchFormatRupiah(largeDataset, 1000);

The caching system was removed because it added complexity and memory overhead. In practice, most applications don't need it. If you really need caching, you can implement a simple cache wrapper:

import { formatRupiah } from '@devnovaa-id/new-rupiah-formatter';

// Simple cache implementation
const createCachedFormatter = (options) => {
  const cache = new Map();
  return (value) => {
    const key = `${value}_${JSON.stringify(options)}`;
    if (cache.has(key)) {
      return cache.get(key);
    }
    const result = formatRupiah(value, options);
    cache.set(key, result);
    return result;
  };
};

// Usage
const cachedFormat = createCachedFormatter({ precision: 0 });
console.log(cachedFormat(1000)); // "Rp1.000" (caches result)
console.log(cachedFormat(1000)); // "Rp1.000" (from cache)

However, in most cases, the formatter is fast enough that caching isn't necessary.
